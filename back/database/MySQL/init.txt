CREATE DATABASE UberVegan;
USE UberVegan;

CREATE TABLE T_ROLE_ROL(
	rol_id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
	rol_name VARCHAR(25)
);

CREATE TABLE T_USER_USR(
	usr_id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
	usr_email VARCHAR(25),
	usr_password VARCHAR(25),
	usr_status INT(2),
	rol_id INT,
        FOREIGN KEY T_USER_USR(rol_id) REFERENCES T_ROLE_ROL(rol_id)
);
