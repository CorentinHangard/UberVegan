CREATE DATABASE UberVegan;
USE UberVegan;
CREATE TABLE T_ROLE_ROL(rol_id INT PRIMARY KEY NOT NULL IDENTITY(1, 1),rol_name VARCHAR(25));
CREATE TABLE T_USER_USR(usr_id INT PRIMARY KEY NOT NULL IDENTITY(1, 1),usr_email VARCHAR(25),usr_password VARCHAR(25),usr_status INT,rol_id INT FOREIGN KEY REFERENCES T_ROLE_ROL(rol_id));
CREATE TABLE refresh_token (id INT PRIMARY KEY NOT NULL IDENTITY(1, 1),refresh_token VARCHAR(255) NOT NULL,usr_id INT FOREIGN KEY REFERENCES T_USER_USR(usr_id));
